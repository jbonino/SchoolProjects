Template Method

Forces


Create a bunch of similar objects that differ slightly.


Solution


Provide a class that contains a method that provides the steps of the algorithm. It allows subclasses to override some of the methods.

Example (Pseudo Code)

abstract class Burger(){
     final makeBurger(){ //so subclasses cant change this method.
          cutBun();
          if(wantsMeat) addMeat();
          if(wantsCheese) addCheese();
          if(wantsVeg) addVeg();
          wrapBurger();
     }

     abstract void addMeat();
     abstract void addCheese();
     abstract void addVeg();

     void cutBun(){ //do cutting! }
     void wrapBurger(){ //wrap it! }

     Boolean wantsMeat(){ return true; }
     Boolean wantsCheese(){ return true; }
     Boolean wantsVeg(){ return true; }
}

class vegBurger(){
     Boolean wantsMeat(){ return false; }
     void addMeat(){}
//still needs to be defined but does nothing      void addCheese(){//add cheese!}
     void addVeg(){//add veges!}
}

class regularBurger(){
     void addMeat(){//add meat!}
     void addCheese(){//add cheese!}
     void addVeg(){//add veges!}
}
