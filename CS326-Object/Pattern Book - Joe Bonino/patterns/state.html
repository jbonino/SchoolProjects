State

Forces


A single object has to behave differently under certain circumstances.


Solution


Allow the object to alter its behavior when its internal state changes. The object will appear to change its class.


Related Patters


  • Singleton
  • Fly Weight


Example (Pseudo Code)

interface HumanState{
     void sharkAttack();
     void holyWater();
     void zombieInfection();
}

class Human{
     HumanState alive, dead, zombie, currentState;

     Human(){
          alive = new Alive(this);
          dead = new Dead(this);
          zombie = new Zombie(this);
          currentState = alive;
     }

     void setHumanState(HumanState state){ currentState = state; }
     void sharkAttack(){ currentState.sharkAttack(); }
     void holyWater(){ currentState.holyWater(); }
     void zombieInfection() { currentState.zombieInfection(); }

     getAliveState(){ return alive; }
     getDeadState(){ return dead; }
     getZombieState(){ return zombie; }
}

class Alive implements HumanState{
     Human human;
     Alive(HumanState newHuman){ human = newHuman; }

     void sharkAttack(){
          human.setHumanState(human.getDeadState());
     }
     void holyWater(){}

     void zombieInfection(){
     human.setHumanState(human.getZombieState());
     }
}

class Dead implements HumanState{
     Human human;
     Dead(HumanState newHuman){ human = newHuman; }

     void sharkAttack(){}

     void holyWater(){
     human.setHumanState(human.getAliveState());
     }
     void zombieInfection(){
          human.setHumanState(human.getZombieState());
     }
}

class Zombie implements HumanState{
     Human human;
     Zombie(HumanState newHuman){ human = newHuman; }

     void sharkAttack(){
     human.setHumanState(human.getDeadState());
     }
     void holyWater(){
     human.setHumanState(human.getAliveState());
     }
     void zombieInfection(){}
}
