Chain of Responsibility

Forces


A need to solve a problem and one of the many object we have can solve it.


Solution


Connect all the objects with a chain and pick one to start solving problem. If that object cant solve it
then it passes it down to the next object on the chain.


Related Patters


  • Composite
  • Command
  • Observer


Example (Java)

interface mathChain(){
     void setNext(mathChain c);
     void calculate(int one,int two,string calculation);
}

class addNums implements mathChain{
     mathChain nextChain;
     void setNext(mathChain c){ nextChain=c; }

     void calculate(int one,int two,string calculation){
          if(calculation=="add"){ System.out.print(one.toString() + " + " + two.toString() + " = " + (one+two).toString);
          else{ nextChain.calculate(one,two,calculation); }
     }
}

class subNums implements mathChain{
     mathChain nextChain;
     void setNext(mathChain c){ nextChain=c; }

     void calculate(int one,int two,string calculation){
          if(calculation=="sub"){ System.out.print(one.toString() + " - " + two.toString() + " = " + (one-two).toString);
          else{ nextChain.calculate(one,two,calculation); }
     }
}

class divNums implements mathChain{
     mathChain nextChain;
     void setNext(mathChain c){ nextChain=c; }

     void calculate(int one,int two,string calculation){
          if(calculation=="div"){ System.out.print(one.toString() + " / " + two.toString() + " = " + (one/two).toString);
          else{ nextChain.calculate(one,two,calculation); }
     }
}

class multNums implements mathChain{
     mathChain nextChain;
     void setNext(mathChain c){ nextChain=c; }

     void calculate(int one,int two,string calculation){
          if(calculation=="mult"){ System.out.print(one.toString() + " * " + two.toString() + " = " + (one*two).toString);
          else{ System.out.print("Incorrect format"); }
     }
}

class test{
     int main(){
     mathChain add = new addNums();//initialize objects
     mathChain sub = new subNums();
     mathChain div = new divNums();
     mathChain mult = new multNums();

     add.setNext(sub);//set up chain
     sub.setNext(div);
     div.setNext(mult);

     add.calculate(10,2,"div"); //prints 10 / 2 = 5
     }
}
